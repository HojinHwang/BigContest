import urllib

url = 'http://find.mk.co.kr/new/search.php?pageNum=1&cat=&cat1=&media_eco=&pageSize=10&sub=all&dispFlag=OFF&page=news&s_kwd=NS+%C8%A8%BC%EE%C7%CE&s_page=news&go_page=&ord=1&ord1=1&ord2=0&s_keyword=NS+%C8%A8%BC%EE%C7%CE&s_i_keyword=NS+%C8%A8%BC%EE%C7%CE&s_author=&y1=2020&m1=01&d1=01&y2=2020&m2=08&d2=25&media%5B1%5D=00001&ord=1&area=ttbd'
driver = webdriver.Chrome('C:/Users/wndms/Downloads/chromedriver.exe')
driver.get(url)
page_num = 1
news = pd.DataFrame()
i = 3
while True:
    try:
        driver.find_element_by_xpath(f'/html/body/center/table/tbody/tr[1]/td[1]/div[{i}]/span[1]/a')
    except:
        try:
            driver.find_element_by_xpath('/html/body/center/table/tbody/tr/td[1]/table/tbody/tr/td/div[2]/a[12]')
        except:
            try:
                driver.find_element_by_xpath('/html/body/center/table/tbody/tr/td[1]/table/tbody/tr/td/div[2]/a[11]')
            except:
                break
            else:
                page_num+=1
                driver.find_element_by_xpath(f'/html/body/center/table/tbody/tr/td[1]/table/tbody/tr/td/div[2]/a[{page_num}]').click()
                i = 3
        else:
            page_num+=1
            clk = page_num%10
            if clk == 0:
                driver.find_element_by_xpath('/html/body/center/table/tbody/tr/td[1]/table/tbody/tr/td/div[2]/a[11]').click()
                i=3
            elif clk ==1:
                driver.find_element_by_xpath('/html/body/center/table/tbody/tr/td[1]/table/tbody/tr/td/div[2]/a[12]').click()
                i=3
            else:
                driver.find_element_by_xpath(f'/html/body/center/table/tbody/tr/td[1]/table/tbody/tr/td/div[2]/a[{clk+1}]').click()
                i=3
    else:
        driver.find_element_by_xpath(f'/html/body/center/table/tbody/tr[1]/td[1]/div[{i}]/span[1]/a').click()
        driver.switch_to_window(driver.window_handles[1])
        temp_title = pd.Series(driver.find_element_by_class_name('top_title').text)
        temp_time = pd.Series(driver.find_element_by_class_name('lasttime').text)
        temp_content = pd.Series(driver.find_element_by_class_name('art_txt').text)
        temp_news = pd.concat([temp_title,temp_time,temp_content],axis=1)
        news = pd.concat([news,temp_news],axis=0).reset_index(drop=True)
        driver.close()
        driver.switch_to_window(driver.window_handles[0])
        i+=1
driver.close()

from konlpy.tag import Okt
okt = Okt()
covid_noun = []
covid_arr = pd.Series()
for idx,i in enumerate(news_20['content']):
    arr_temp = []
    temp_pos = okt.pos(i)
    for name, tag in temp_pos:
        if tag == 'Noun':
            if len(name)>1:
                covid_noun.append(name)
                arr_temp.append(name)
    covid_arr.loc[idx] = arr_temp
covid_set = {}
for word in covid_noun:
    covid_set[word] = covid_set.get(word, 0) + 1
